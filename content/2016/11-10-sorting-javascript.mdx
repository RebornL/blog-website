## еҶ’жіЎжҺ’еәҸ

```javascript
function bubbleSort(arr) {
    var len = arr.length;
    for (var i = 0; i < len; i++) {
        for (var j = 1; j < len - i; j++) {
            if (arr[j - 1] > arr[j]) {
                var temp = arr[j - 1];
                arr[j - 1] = arr[j];
                arr[j] = temp;
            }
        }
    }
    return arr;
}
```

## йҖүжӢ©жҺ’еәҸ

```javascript
function selectSort(arr) {
    var len = arr.length;
    for (var i = 0; i < len - 1; i++) {
        var min = i;
        for (var j = i + 1; j < len; j++) {
            if (arr[min] > arr[j]) min = j;
        }
        if (min !== i) {
            var temp = arr[i];
            arr[i] = arr[min];
            arr[min] = temp;
        }
    }
    return arr;
}
```

```javascript
function insertionSort(arr) {
    var len = arr.length;
    for (var i = 1; i < len; i++) {
        if (arr[i - 1] > arr[i]) {
            var temp = arr[i];
            var index = i;
            for (var j = i - 1; j >= 0; j--) {
                if (arr[j] > temp) {
                    arr[j + 1] = arr[j];
                    index = j;
                }
            }
            arr[index] = temp;
        }
    }
    return arr;
}
```
